import { fromJS } from 'immutable';
import reducerHandler from 'shared/utils/reducerHandler';
import { formatDate, mapToDateFormat } from 'shared/utils/DateTimeFormat';
import handleCalendarEvents from '../utils/calendarEvents';
import {
  SET_RESOURCE_IDS,
  SET_RESOURCE_SELECTED_DATE,
  FETCH_RESOURCES_BOOKING_SUCCESS,
  FETCH_RESOURCES_BOOKING_FAILURE,
  DELETE_RESOURCES_INFO,
  WINDOW_RESIZE,
  SEND_EMPTY_REQUEST,
  SYNC_DATA_FROM_BOOKING_INFO_TO_CALENDAR,
  FETCH_READY4CHECKOUT_SUCCESS,
  FETCH_INCART_SUCCESS
} from '../actions/resourceBooking';
import { START_EVENT_DATE, DATE_FORMAT } from '../components/FullCalendar/utils';
 
const initialState = fromJS({
  resource_ids: [],
  // selected_date: formatDate(START_EVENT_DATE, mapToDateFormat(DATE_FORMAT)),
  start_date: formatDate(START_EVENT_DATE, mapToDateFormat(DATE_FORMAT)),
  end_date: formatDate(START_EVENT_DATE, mapToDateFormat(DATE_FORMAT)),
  include_linked_resources: true,
  resourceInfo: [],
  bookingInfo: [],
  resize: false,
  ready4Checkout: false, // if current receipt is valid to make payment
  inCart: false // reservation has added into shopping cart
});
 
const handlers = {
 
  [SYNC_DATA_FROM_BOOKING_INFO_TO_CALENDAR](state, { payload: { value } }) {
    let bookingInfo = state.get('bookingInfo');
 
    if (bookingInfo) {
      // keep bookings which were not in booking information panel
      bookingInfo = bookingInfo.filter(item =>
        // permit id will be -1 for new reservation workflow, and prop current event is undefined
        // current event will be true for modify reservation workflow
        (item.get('bookingAssignment') !== 0 && item.get('permitID') !== -1 && !item.get('currentEvent')) ||
        // keep booking that is not belongs to current user
        item.get('ownerPendingReceipt') === false
      );
    } else {
      bookingInfo = fromJS([]);
    }
 
    return state.set('bookingInfo', bookingInfo.concat(fromJS(value)));
  },
 
  [SET_RESOURCE_IDS](state, { payload: { value } }) {
    return state.withMutations((s) => {
      s.set('resource_ids', fromJS(value));
    });
  },
 
  [DELETE_RESOURCES_INFO](state, { payload: { value } }) {
    return state.withMutations((s) => {
      s.set('resourceInfo', fromJS(value));
    });
  },
 
  [SET_RESOURCE_SELECTED_DATE](state, { payload: { value } }) {
    return state.withMutations((s) => {
      s.set('start_date', value);
      s.set('end_date', value);
    });
  },
 
  [FETCH_RESOURCES_BOOKING_SUCCESS](state, { payload: { body: { booking_list: bookingList } } }) {
    if (bookingList) {
      // all data below is came from database
      const { resourceInfo, bookingInfo } = bookingList;
      const resources = [];
      const bookings = [];
 
      resourceInfo && resourceInfo.forEach((resource) => {
        /**
         * In shared/Fullcalendar component:
         * After "this.transferCalling('setResources', this.props.resources)":
         * Full-calendar will auto add a id key base on the config's 'resourceIdField'
         * If we don't give a id to resource here:
         * Resources always not equal this.props.resources
         */
        const eventResource = Object.assign({}, resource, {
          id: resource.resourceID
        });
        resources.push(eventResource);
      });
 
      bookingInfo && bookingInfo.forEach((booking) => {
        const eventBooking = handleCalendarEvents(booking);
        bookings.push(eventBooking);
      });
 
      return state.withMutations((s) => {
        s.set('resourceInfo', fromJS(resources));
        s.set('bookingInfo', fromJS(bookings));
      });
    }
    return state;
  },
 
  [FETCH_RESOURCES_BOOKING_FAILURE](state) {
    return state.withMutations((s) => {
      s.set('resource_ids', fromJS([]));
    });
  },
 
  [WINDOW_RESIZE](state, { payload: { value } }) {
    return state.set('resize', value);
  },
 
  [SEND_EMPTY_REQUEST](state) {
    return state.withMutations((s) => {
      s.set('resourceInfo', fromJS([]));
    });
  },
 
  [FETCH_READY4CHECKOUT_SUCCESS](state, { payload: { body } }) {
    return state.set('ready4Checkout', body.ready4checkout === 'true');
  },
 
  [FETCH_INCART_SUCCESS](state, { payload: { body } }) {
    return state.set('inCart', body.in_cart === 'true');
  }
};
 
export default reducerHandler(initialState, handlers);
